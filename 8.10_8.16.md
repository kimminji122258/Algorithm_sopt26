<h4>알고리즘 12강 실습문제</h4>

* 스택을 이용한 중위표현 수식 --> 후위표현 수식

```python
class ArrayStack:

    def __init__(self):
        self.data = []

    def size(self):
        return len(self.data)

    def isEmpty(self):
        return self.size() == 0

    def push(self, item):
        self.data.append(item)

    def pop(self):
        return self.data.pop()

    def peek(self):
        return self.data[-1]

prec = {
    '*': 3, '/': 3,
    '+': 2, '-': 2,
    '(': 1
}

def solution(S):
    opStack = ArrayStack()
    answer = ''
    
    for i in S:
        if i in '(':
            opStack.push(i)
        elif i in ')':
            while not opStack.isEmpty():
                if opStack.peek() == '(':
                    opStack.pop()
                    break
                else:
                    answer += opStack.pop()
        elif i in '*/+-':
            while opStack.size() > 0 and prec[opStack.peek()] >= prec[i]:
                answer += opStack.pop()
            opStack.push(i)    
                
        # 피연산자라면
        else:
                answer += i
                
    while not opStack.isEmpty():
        answer += opStack.pop()
    
    return answer
```



<h4>알고리즘 13강 실습문제</h4>

* 후위표현식을 스택을 이용해 중위표현식으로 바꾸고, 계산해라

```python
class ArrayStack:

    def __init__(self):
        self.data = []

    def size(self):
        return len(self.data)

    def isEmpty(self):
        return self.size() == 0

    def push(self, item):
        self.data.append(item)

    def pop(self):
        return self.data.pop()

    def peek(self):
        return self.data[-1]

# 문자열을 받아, 리스트로 만드는 함수
def splitTokens(exprStr):
    tokens = []
    val = 0
    valProcessing = False
    for c in exprStr:
        if c == ' ':
            continue
        if c in '0123456789':
            val = val * 10 + int(c)
            valProcessing = True
        else:
            if valProcessing:
                tokens.append(val)
                val = 0
            valProcessing = False
            tokens.append(c)
    if valProcessing:
        tokens.append(val)

    return tokens

# 중위 표현법을 후위 표현법으로 바꾸는 함수
def infixToPostfix(tokenList):
    prec = {
        '*': 3,
        '/': 3,
        '+': 2,
        '-': 2,
        '(': 1,
    }

    opStack = ArrayStack()
    postfixList = []

    for i in tokenList:
        if i == '(':
            opStack.push(i)
        elif i == ')':
            while not opStack.isEmpty():
                if opStack.peek() == '(':
                    opStack.pop()
                    break
                else:
                    postfixList.append(opStack.pop())
        # 피연산자라면            
        elif type(i) is int:
            postfixList.append(i) 
                
        else:
            while opStack.size() > 0 and prec[opStack.peek()] >= prec[i]:
                postfixList.append(opStack.pop())
            opStack.push(i)       
                
    while not opStack.isEmpty():
        postfixList.append(opStack.pop())
        
    return postfixList

# 후위 표현법을 계산하는 함수
def postfixEval(tokenList):
    valStack = ArrayStack()
    
    for t in tokenList:
        if type(t) is int:
            valStack.push(t)
        elif t == '+':
            valStack.push(valStack.pop() + valStack.pop())
        elif t == '-':
            firstOut = valStack.pop()
            laterOut = valStack.pop()
            valStack.push(laterOut - firstOut)
        elif t == '*':
            valStack.push(valStack.pop() * valStack.pop())
        elif t == '/':
            firstOut = valStack.pop()
            laterOut = valStack.pop()
            valStack.push(laterOut / firstOut)
    
    return valStack.pop()
    

def solution(expr):
    tokens = splitTokens(expr)
    postfix = infixToPostfix(tokens)
    val = postfixEval(postfix)
    return val
```



<h4>프로그래머스 level-2</h4>

* 다리를 지나는 트럭

```python
def solution(bridge_length, weight, truck_weights):
    queue = [0] * bridge_length
    time = 0    
    sum = 0
    
    while queue:
        time += 1
        sum -= queue.pop(0)
        
        if truck_weights:
            if sum + truck_weights[0] <= weight:
                sum += truck_weights[0]
                queue.append(truck_weights.pop(0))
            else:
                queue.append(0)
    
    return time
```
